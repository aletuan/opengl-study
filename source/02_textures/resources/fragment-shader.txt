#version 330 core

in vec3 ourColor;
in vec2 TexCoord;

out vec4 color;

void main() {
    //set every drawn pixel to white
    //color = vec4(ourColor, 1.0f);
    //color = texture(ourTexture, TexCoord);
    //we can mix resulting texture color with vertex color
    color = texture(ourTexture1, TexCoord) * vec4(ourColor, 1.0f);
}
